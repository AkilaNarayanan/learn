def drug_classify(drugname):
    for i in drug_suffix.keys():
        if drugname.endswith(i):
           ## print(True)
            return drug_suffix[i]

def get_sendiment_label(text):
    blob=TextBlob(text)
    if blob.polarity>0:
        result='positive'
    elif blob.polarity<0:
        result='negative'
    else:
        result='neutral'
    return result
